23       * @see LicenseOutput#output(java.util.Map)

25      public void output(final Map<String, List<License>> entries) {
26          final Set<String> keySet = entries.keySet();
27          final Set<String> knownEntries = new LinkedHashSet<String>();
28          final Set<String> unknownEntries = new LinkedHashSet<String>();
29  
30          for (String entry : keySet) {
31              final List<License> licenses = entries.get(entry);
32              if(licenses.size() > 1) {
33                  final StringBuilder sb = new StringBuilder("");
34                  for(License licence : licenses) {
35                      sb.append(String.format("%s(%s) ", licence.getName(), licence.getUrl()));
36                  }
37                  knownEntries.add(String.format("%s has multiple licenses: %s", entry, sb));
38              } else if(licenses.size() == 1) {
39                  final License licence = licenses.get(0);
40                  knownEntries.add(String.format("%s %s (%s)", entry, licence.getName(), licence.getUrl()));
41              } else {
42                  unknownEntries.add(entry);
43              }
44          }
45  
46          if(unknownEntries.size() == 0) {
47              log.info("All artifact entries have associated license information.");
48          } else {
49              log.warn("Some of the entries have no associated license information or the plugin wasn't able to determine them. Please check them manually.");
50          }
51  
52          log.info("");
53          log.info("The archive contains following entries with known license information:");
54          for (String entryDesc : knownEntries) {
55              log.info(entryDesc);
56          }
57  
58          if(unknownEntries.size() > 0) {
59              log.info("");
60              log.warn("The archive contains following entries with uknown license inforamtion:");
61              for (String archiveName : unknownEntries) {
62                  log.warn(archiveName);
63              }
64          }
65      }
