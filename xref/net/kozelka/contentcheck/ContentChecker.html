28      /**
29       * Checks a content of {@code sourceFile} according to an allowed content defined by {@code listingFile}.
30       * 
31       * @param listingFile a file that defines allowed content
32       * @param sourceFile directory or archive file to be checked
33       * 
34       * @return the result of source check
35       * 
36       * @throws IOException if something very bad happen
37       */
38      public CheckerOutput check(final File listingFile, final File sourceFile) throws IOException{
39          final Set<String> approvedEntries = readListing(listingFile);
40          final int totalCount = introspector.readEntries(sourceFile);
41          //XXX dagi: duplicit entries detection https://github.com/pkozelka/contentcheck-maven-plugin/issues#issue/4
42          final Set<String> actualEntries = introspector.getEntries();
43          events.fire.summary(sourceFile, actualEntries.size(), totalCount);
44          return new CheckerOutput(approvedEntries, actualEntries);
45      }
46  
47      protected Set<String> readListing(final File listingFile) throws IOException {
48          final Set<String> expectedPaths = new LinkedHashSet<String>();
49          final BufferedReader reader = new BufferedReader(new FileReader(listingFile));
50          try {
51              int totalCnt = 0;
52              String line;
53              while ((line = reader.readLine())!= null) {
54                  totalCnt ++;
55                  line = line.trim();
56                  final boolean ignoreLine = line.length() == 0 || line.startsWith("#");// we ignore empty and comments lines
57                  if (!ignoreLine) { 
58                      if(expectedPaths.contains(line)) {
59                          events.fire.duplicate(listingFile, line);
60                      }
61                      expectedPaths.add(line);
62                  } 
63              }
64              events.fire.contentListingSummary(listingFile, expectedPaths.size(), totalCnt);
65              return expectedPaths;
66          } finally {
67              reader.close();
68          }
69      }
70  
71      public static interface Events {
72          void summary(File sourceFile, int checkedCount, int totalCount);
73  
74          void duplicate(File listingFile, String line);
75  
76          void contentListingSummary(File listingFile, int pathCount, int totalCount);
77      }
