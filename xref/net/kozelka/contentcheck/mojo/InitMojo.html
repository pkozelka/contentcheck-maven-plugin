31  /**
32   * Convenient way to start using contentcheck plugin, with all its capabilities enabled.
33   * Use it to make all following at once:
34   * <ul>
35   *     <li>generate the <code>approved-content.txt</code> file to match current war content</li>
36   *     <li>count war overlaps</li>
37   *     <li>TODO: show resolution hints</li>
38   *     <li>generate POM fragment with complete configuration (under target/contentcheck-maven-plugin/)</li>
39   *     <li>operate on each module of multi-module project</li>
40   *     <li>TODO: optionally, adjust POM configuration: `-DeditPom=true`</li>
41   * </ul>
42   */
43  @Mojo(name = "init")
44  public class InitMojo extends AbstractMojo {
45  
46      @Parameter(defaultValue = "${project.build.directory}/contentcheck-maven-plugin")
47      File outputDirectory;
48  
49      /**
50       * The Maven Project.
51       */
52      @Parameter(property = "project", required = true, readonly = true)
53      MavenProject project;
54  
55      private boolean reportJarPairs = true;
56  
57      public void execute() throws MojoExecutionException, MojoFailureException {
58          if (! project.getPackaging().equals("war")) {
59              getLog().warn(String.format("Skipping '%s' module - currently we support only war", project.getPackaging()));
60              return;
61          }
62          final Build build = project.getBuild();
63          final File sourceFile = new File(build.getDirectory(), build.getFinalName() + "." + project.getArtifact().getType());
64          try {
65              // report class overlaps / conflicts
66              final ClassConflictAnalyzer ccd = new ClassConflictAnalyzer();
67              final List<ArchiveInfo> archives = ArchiveLoader.loadWar(sourceFile);
68              final ClassConflictReport report = ccd.analyze(archives);
69              final StreamConsumer output = new StreamConsumer() {
70                  public void consumeLine(String line) {
71                      getLog().error(line);
72                  }
73              };
74              if (reportJarPairs) {
75                  final ClassConflictPrinter printer = new ClassConflictPrinter();
76                  printer.setPreviewThreshold(5);
77                  printer.setOutput(output);
78                  printer.print(report);
79              }
80  
81              // list source entries into `approved-content.txt`
82              outputDirectory.mkdirs();
83              final File contentListing = new File(outputDirectory, "approved-content.txt");
84              final List<ActualEntry> sourceEntries = scanActualEntries(sourceFile);
85              getLog().info(String.format("Generated %d entries.", sourceEntries.size()));
86  //            getLog().info(String.format("The source contains %d entries, but only %d matches the plugin configuration criteria.", count, sourceEntries.size()));
87              ExpectUtils.generateListing(sourceEntries, contentListing);
88              getLog().info(String.format("The listing file '%s' has been successfully generated.", contentListing));
89  
90              // generate pom fragment
91              final String template = IOUtil.toString(InitMojo.class.getResourceAsStream("init-pom-template.txt"));
92              final String version = readMyOwnVersion();
93              String fragment = template.replace("@overlap.count@", report.getTotalOverlaps() + "");
94              fragment = fragment.replace("@project.version@", version); //TODO - find and use real version here!
95              final File fragmentFile = new File(outputDirectory, "fragment-pom.xml");
96              FileUtils.fileWrite(fragmentFile, fragment);
97          } catch (IOException e) {
98              throw new MojoFailureException(e.getMessage(), e);
99          }
100     }

102     // TODO: following methods should be moved to/unified with ContentChecker class
103     private List<ActualEntry> scanActualEntries(File sourceFile) throws IOException {
104 //        getLog().warn(String.format("Archives of vendor '%s', indicated by manifest entry '%s', will not be added to the list", vendorId, manifestVendorEntry));
105         final ContentIntrospector introspector = VendorFilter.createIntrospector(new MyIntrospectionListener(getLog()),
106             true, project.getGroupId(), VendorFilter.DEFAULT_VENDOR_MANIFEST_ENTRY_NAME, "**/*.jar");
107         getLog().info("Scanning " + sourceFile);
108         introspector.setSourceFile(sourceFile);
109         introspector.getEvents().addListener(new MyIntrospectionListener(getLog()));
110         final List<ActualEntry> actualEntries = new ArrayList<ActualEntry>();
111         final ContentIntrospector.Events collector = new ContentCollector(actualEntries);
112         introspector.getEvents().addListener(collector);
113         introspector.walk();
114         return actualEntries;
115     }

117     private static Properties readPomProperties() throws IOException {
118         final Properties result = new Properties();
119         final String resource = "META-INF/maven/net.kozelka.maven/contentcheck-maven-plugin/pom.properties";
120         final InputStream is = InitMojo.class.getClassLoader().getResourceAsStream(resource);
121         if (is == null) {
122             throw new FileNotFoundException("classpath:" + resource);
123         }
124         try {
125             result.load(is);
126             return result;
127         } finally {
128             is.close();
129         }
130     }
131 
132     private static String readMyOwnVersion() throws IOException {
133         final Properties pomProperties = readPomProperties();
134         final String version = pomProperties.getProperty("version");
135         if (version == null) {
136             throw new IllegalStateException("Failed to load plugin's own version");
137         }
138         return version;
